# camoufox is a Firefox-based anti-detect browser: it spoofs the whole
# fingerprint (WebGL, canvas, fonts, navigator, screen). Its browser binary is
# fetched at runtime by run.sh (`python -m camoufox fetch`).
#
# playwright MUST be pinned to the version whose bundled Firefox matches the
# Camoufox browser build, or the Juggler protocol mismatches (e.g. the
# "setDefaultViewport ... isMobile" error). Camoufox 0.4.x ships Firefox 135,
# which pairs with Playwright 1.51.0 — keep this in sync with the base image
# tag in build.yaml / Dockerfile. Without the pin, camoufox pulls the latest
# Playwright and the launch fails.
aiohttp>=3.9
playwright==1.51.0
camoufox>=0.4.11
